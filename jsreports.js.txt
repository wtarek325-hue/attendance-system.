// التحقق من تجاوز أيام العمل
function checkOvertime(employeeId, month) {
    const attendance = JSON.parse(localStorage.getItem('attendance') || '[]');
    const monthAttendance = attendance.filter(a => 
        a.employeeId === employeeId && 
        a.date.startsWith(month)
    );
    
    const daysInMonth = new Date(month + '-01').getDate();
    const attendanceDays = monthAttendance.length;
    
    if (attendanceDays > daysInMonth) {
        return `تحذير: الموظف تجاوز العدد المسموح (${attendanceDays}/${daysInMonth})`;
    }
    
    return `طبيعي (${attendanceDays}/${daysInMonth})`;
}
